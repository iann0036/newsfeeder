Class SheetCollection

Provides a storage class for Sheets that belong to a user

$sheets = new SheetCollection();

$sheets->add('New Sheet');

$sheets->remove(2);

foreach ($sheets as $sheet) {
    echo $sheet['id'];
    echo $sheet['name'];
    echo $sheet['updated'];
}
