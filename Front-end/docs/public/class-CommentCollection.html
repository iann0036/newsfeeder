Class CommentCollection

Provides a storage class for Comments on a particluar item

$comments = new CommentCollection(1);

foreach($comments as $comment) {
    echo $comment['author'];
    echo $comment['via'];
    echo $comment['date'];
    echo $comment['comment'];
    echo '<img src="'.$comment['author'].'" />';
}
