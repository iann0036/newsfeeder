Class User

Identifies a particular user of the service

$user = new User();
$otheruser = new User('jbloggs','Joe Bloggs',1,0,'joe@example.com');

if ($user->validUser()) {
    echo $user->getUsername();
    echo $user->getName();
    echo $user->getEmail();
    if ($user->isAdmin())
        echo 'User is an Administrator';
    if ($user->isAltDesign())
        echo $user->isAltDesign();
}
